# DOWNLOAD HUMAN TRANSCRIPTOME REFERENCE
#
# Why: Salmon needs to know all human transcript sequences to quantify gene expression
# This downloads the GENCODE human transcriptome (same source as paper used)
#
# What it downloads:
# - All human transcript sequences (cDNA)
# - ~250,000 transcripts from ~60,000 genes
# - Size: ~50 MB compressed, ~200 MB uncompressed
#
# The paper used Ensembl annotations - GENCODE is the official Ensembl human gene set
#
# Code formatting and debugging assisted by Claude AI (Anthropic, 2026)
 
# Set directories
BASE_DIR="/scratch/alice/t/tlti1/altpipeline"
REF_DIR="$BASE_DIR/reference"
 
echo "=== Downloading Human Transcriptome Reference ==="
echo ""
 
# Download GENCODE human transcriptome (release 46 - latest)
# Using the same comprehensive transcript set the paper would have used
echo "Downloading GENCODE v46 human transcriptome..."
cd $REF_DIR
 
wget -c https://ftp.ebi.ac.uk/pub/databases/gencode/Gencode_human/release_46/gencode.v46.transcripts.fa.gz
 
echo ""
echo "✓ Download complete"
 
# Check file size
FILE_SIZE=$(ls -lh gencode.v46.transcripts.fa.gz | awk '{print $5}')
echo "File size: $FILE_SIZE"

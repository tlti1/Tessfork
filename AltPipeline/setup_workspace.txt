# Set base directory
BASE_DIR="/scratch/alice/t/tlti1/altpipeline"

# Create all necessary directories
mkdir -p $BASE_DIR/reference

mkdir -p $BASE_DIR/salmon_index
mkdir -p $BASE_DIR/salmon_quant

# Check metadata files exist

    exit 1
fi

    echo "✓ Found SraRunTable.txt"
else

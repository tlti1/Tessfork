#!/bin/bash
# File: build_salmon_index.sh
# Purpose: Download human transcriptome and build Salmon index

cd ~/organoid_project/reference

# Download GENCODE human transcriptome release 46
wget https://ftp.ebi.ac.uk/pub/databases/gencode/Gencode_human/release_46/gencode.v46.transcripts.fa.gz

# Decompress
gunzip gencode.v46.transcripts.fa.gz

# Build Salmon index
# k-mer length = 31 (default)
# Note: Warnings about missing decoy sequences are expected and acceptable
salmon index -t gencode.v46.transcripts.fa -i salmon_index -k 31

echo "Salmon index built successfully"
ls -lh salmon_index/

SCRIPT 4: Main Salmon Quantification Pipeline (SLURM Job Array)
================================================================================
#!/bin/bash
#SBATCH --job-name=salmon_quant
#SBATCH --array=1-16
#SBATCH --time=02:00:00
#SBATCH --mem=8G
#SBATCH --cpus-per-task=4
#SBATCH --output=../logs/salmon_%A_%a.out
#SBATCH --error=../logs/salmon_%A_%a.err

# File: salmon_pipeline.sh
# Purpose: Download FASTQ, run Salmon quantification, cleanup
# Usage: sbatch salmon_pipeline.sh (array range set by batch script)

# Load required modules
module load salmon/1.10.2-5pennjh
module load sratoolkit/3.0.0-5fetwpi

# Define directories
PROJECT_DIR=~/organoid_project
FASTQ_DIR=${PROJECT_DIR}/fastq
SALMON_INDEX=${PROJECT_DIR}/reference/salmon_index
OUTPUT_DIR=${PROJECT_DIR}/salmon_output

# Get SRR accession number for this array task
SRR=$(sed -n "${SLURM_ARRAY_TASK_ID}p" ${PROJECT_DIR}/all_734_SRR_list.txt)

echo "========================================"
echo "Processing ${SRR}"
echo "Array task: ${SLURM_ARRAY_TASK_ID}"
echo "Start time: $(date)"
echo "========================================"

# Download paired-end FASTQ files from SRA
echo "Downloading ${SRR}..."
cd ${FASTQ_DIR}
fasterq-dump --split-files ${SRR} --threads 4

# Verify download succeeded
if [ ! -f "${SRR}_1.fastq" ]; then
    echo "ERROR: Download failed for ${SRR}"
    exit 1
fi

echo "Download complete for ${SRR}"

# Run Salmon quantification
echo "Running Salmon quantification on ${SRR}..."
salmon quant \
    -i ${SALMON_INDEX} \
    -l A \
    -1 ${FASTQ_DIR}/${SRR}_1.fastq \
    -2 ${FASTQ_DIR}/${SRR}_2.fastq \
    -p 4 \
    --validateMappings \
    -o ${OUTPUT_DIR}/${SRR}

# Check if Salmon completed successfully
if [ $? -eq 0 ]; then
    echo "Salmon quantification completed successfully for ${SRR}"
    
    # Delete FASTQ files to save disk space
    rm ${FASTQ_DIR}/${SRR}_1.fastq ${FASTQ_DIR}/${SRR}_2.fastq
    echo "Cleaned up FASTQ files for ${SRR}"
else
    echo "ERROR: Salmon quantification failed for ${SRR}"
    exit 1
fi

echo "========================================"
echo "End time: $(date)"
echo "Successfully processed ${SRR}"
echo "========================================"

#!/bin/bash
# File: submit_all_734.sh
# Purpose: Automatically submit all 734 samples in batches of 16
# Note: Starts from sample 17 as samples 1-16 were processed in pilot run

TOTAL=734
BATCH_SIZE=16
START_SAMPLE=17  # Skip first 16 already processed

for START in $(seq $START_SAMPLE $BATCH_SIZE $TOTAL); do
    END=$((START + BATCH_SIZE - 1))
    if [ $END -gt $TOTAL ]; then
        END=$TOTAL
    fi
    
    echo "========================================"
    echo "=== Submitting batch $START-$END ==="
    echo "========================================"
    
    # Update array range in the SLURM script
    sed -i "s/#SBATCH --array=.*/#SBATCH --array=$START-$END/" salmon_pipeline.sh
    
    # Submit batch to SLURM
    JOB_ID=$(sbatch salmon_pipeline.sh | awk '{print $4}')
    echo "Job ID: $JOB_ID"
    
    # Wait for all jobs in this batch to complete
    echo "Waiting for batch to complete..."
    while squeue -u $USER -h | grep -q $JOB_ID; do
        RUNNING=$(squeue -u $USER -h | grep " R " | wc -l)
        PENDING=$(squeue -u $USER -h | grep " PD " | wc -l)
        COMPLETED=$(ls ~/organoid_project/salmon_output/ 2>/dev/null | wc -l)
        echo "  Running: $RUNNING | Pending: $PENDING | Completed: $COMPLETED/$TOTAL"
        sleep 60
    done
    
    echo "Batch $START-$END complete!"
    echo ""
done

echo "=========================================="
echo "ALL 734 SAMPLES PROCESSED SUCCESSFULLY!"
echo "=========================================="
echo "Total samples completed: $(ls ~/organoid_project/salmon_output/ | wc -l)"

#!/bin/bash
# File: extract_timing.sh
# Purpose: Extract timing information for performance benchmarking

cd ~/organoid_project/logs

echo "Extracting processing times from SLURM logs..."
echo "Sample,Start_Time,End_Time" > timing_data.csv

for logfile in salmon_*_*.out; do
    SRR=$(grep "Processing SRR" $logfile | awk '{print $2}')
    START=$(grep "Start time:" $logfile | cut -d' ' -f3-)
    END=$(grep "End time:" $logfile | cut -d' ' -f3-)
    
    if [ ! -z "$SRR" ]; then
        echo "$SRR,$START,$END" >> timing_data.csv
    fi
done

echo "Timing data extracted to timing_data.csv"
head -10 timing_data.csv

# Calculate summary statistics (requires GNU date)
# This shows average processing time per sample
awk -F',' 'NR>1 {print $1}' timing_data.csv | wc -l


================================================================================
TROUBLESHOOTING NOTES
================================================================================

Common Issues Encountered:

1. FILE PATH ERROR (Fixed)
   Problem: Script looked for ../all_734_SRR_list.txt instead of ./all_734_SRR_list.txt
   Solution: sed -i 's|${PROJECT_DIR}/../all_734_SRR_list.txt|${PROJECT_DIR}/all_734_SRR_list.txt|' salmon_pipeline.sh

2. QOS JOB LIMIT
   Problem: Can only submit 16 jobs at once on standard queue
   Solution: Batch submission script waits for each batch to complete before submitting next

3. DISK QUOTA
   Problem: 100GB limit insufficient for storing all FASTQ files
   Solution: Delete FASTQ files immediately after Salmon quantification

4. MODULE LOADING
   Problem: Need to reload modules in SLURM job scripts
   Solution: Always include module load commands at top of job scripts

5. SRA TOOLKIT CONFIGURATION
   Problem: First-time use requires vdb-config setup
   Solution: Run vdb-config --set command before using fasterq-dump
